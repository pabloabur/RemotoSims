13 # calculates the coupling conductance between two compartments
14 # Inputs: cytR. Cytoplasmatic resistance in Ohm.cm
15 # lComp1, lComp2: length of the compartments in mum
16 # dComp1, dComp2 diameter of the compartments in mum
17 # Output: coupling conductance in MS
18 #
19 def calcGCoupling(cytR, lComp1, lComp2, dComp1, dComp2):
20  rAxis1 = (cytR * lComp1) / (math.pi * math.pow(dComp1/2, 2))
21  rAxis2 = (cytR * lComp2) / (math.pi * math.pow(dComp2/2, 2))
22 
23  return (1e2 * 2) / (rAxis1 + rAxis2)
